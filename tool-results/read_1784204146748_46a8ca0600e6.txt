   136→   136→// Helper: ensure a value is always a string
   137→   137→function ensureString(val: any): string {
   138→   138→  if (typeof val === 'string') return val;
   139→   139→  if (Array.isArray(val)) return val.map(v => typeof v === 'string' ? v : v != null ? JSON.stringify(v) : '').filter(Boolean).join('\n');
   140→   140→  if (val != null && typeof val === 'object') return JSON.stringify(val);
   141→   141→  if (val == null) return '';
   142→   142→  return String(val);

   145→   145→// V64: Strip Markdown from display text for clean rendering
   146→   146→function stripDisplayMarkdown(text: string): string {
   147→   147→  if (!text) return text;
   148→   148→  let r = text;
   149→   149→  r = r.replace(/^#{1,6}\s+/gm, '');
   150→   150→  r = r.replace(/\*\*(.+?)\*\*/g, '$1');
   151→   151→  r = r.replace(/__(.+?)__/g, '$1');
   152→   152→  r = r.replace(/\*(.+?)\*/g, '$1');
   153→   153→  r = r.replace(/(?<!\w)_(.+?)_(?!\w)/g, '$1');
   154→   154→  r = r.replace(/^[\-\*]\s+/gm, '');
   155→   155→  r = r.replace(/^[\-\*]{3,}\s*$/gm, '');
   156→   156→  r = r.replace(/`(.+?)`/g, '$1');
   157→   157→  r = r.replace(/\[(.+?)\]\(.+?\)/g, '$1');
   158→   158→  r = r.replace(/\n{3,}/g, '\n\n');
   159→   159→  return r.trim();
   160→   160→}

   162→   162→// V1037 FIX: Split text into readable paragraphs.
   163→   163→// Source article text often comes as ONE big block (no \n newlines),
   164→   164→// so split('\n') produces a single <p> tag with ~1000+ characters — unreadable.
   165→   165→// This helper splits long paragraphs by sentence boundaries.
   166→   166→const MAX_PARAGRAPH_LENGTH = 280;
   167→   167→const SENTENCES_PER_PARAGRAPH = 2;
   168→   168→
   169→   169→function splitIntoParagraphs(text: string): string[] {
   170→   170→  if (!text) return [];
   171→   171→  const trimmed = text.trim();
   172→   172→  if (!trimmed) return [];
   173→   173→
   174→   174→  const rawParagraphs = trimmed
   175→   175→    .split(/\n+/)
   176→   176→    .map(p => p.trim())
   177→   177→    .filter(p => p.length > 0);
   178→   178→
   179→   179→  const result: string[] = [];
   180→   180→  for (const para of rawParagraphs) {
   181→   181→    if (para.length <= MAX_PARAGRAPH_LENGTH) {
   182→   182→      result.push(para);
   183→   183→      continue;
   184→   184→    }
   185→   185→
   186→   186→    // Split by sentence-ending punctuation (. ! ? ؟) followed by optional whitespace
   187→   187→    // and a capital letter or Arabic letter
   188→   188→    // V1069: \s* instead of \s+ — handles LLM output like "presión.El" (no space after period)
   189→   189→    const sentences = para
   190→   190→      .split(/(?<=[.!?؟])\s*(?=[A-ZÀ-ÿ\u0600-\u06FF\u00C0-\u017F«"])/)
   191→   191→      .map(s => s.trim())
   192→   192→      .filter(s => s.length > 0);
   193→   193→
   194→   194→    if (sentences.length <= 1) {
   195→   195→      result.push(para);
   196→   196→      continue;
   197→   197→    }
   198→   198→
   199→   199→    for (let i = 0; i < sentences.length; i += SENTENCES_PER_PARAGRAPH) {
   200→   200→      const group = sentences.slice(i, i + SENTENCES_PER_PARAGRAPH).join(' ');
   201→   201→      if (group.trim().length > 0) {
   202→   202→        result.push(group.trim());
   203→   203→      }
   204→   204→    }
   205→   205→  }
   206→   206→
   207→   207→  return result;
   208→   208→}

   226→   226→function parseAnalysisSections(fullContent: string): AnalysisSection[] {
   227→   227→  if (!fullContent) return [];
   228→   228→
   229→   229→  // Clean the content first
   230→   230→  const cleaned = stripDisplayMarkdown(fullContent);
   231→   231→
   232→   232→  // Check if it has [1]-[6] structure (with or without spaces: [1] or [ 1 ])
   233→   233→  if (!/\[\s*1\s*\]/.test(cleaned)) {
   234→   234→    // No [1]-[6] structure — return as single section
   235→   235→    return [{ num: 0, title: '', content: cleaned }];
   236→   236→  }
   237→   237→
   238→   238→  const sections: AnalysisSection[] = [];
   239→   239→  // Split by [N] markers (with or without spaces)
   240→   240→  const parts = cleaned.split(/\[\s*(\d)\s*\]/);
   241→   241→
   242→   242→  // parts[0] is text before [1], then alternating: number, content
   243→   243→  for (let i = 1; i < parts.length; i += 2) {
   244→   244→    const num = parseInt(parts[i], 10);
   245→   245→    const content = (parts[i + 1] || '').trim();
   246→   246→    if (num >= 1 && num <= 6 && content) {
   247→   247→      // Extract the title from the first line of content
   248→   248→      const lines = content.split('\n');
   249→   249→      const firstLine = lines[0].trim();
   250→   250→      const config = sectionConfig[num];
   251→   251→      // The title is typically the first line, rest is content
   252→   252→      // But sometimes the title is part of the [N] heading and rest is body
   253→   253→      const bodyLines = lines.slice(1).join('\n').trim();
   254→   254→      const sectionTitle = config?.title || firstLine;
   255→   255→      const sectionContent = bodyLines || firstLine;
   256→   256→      sections.push({ num, title: sectionTitle, content: sectionContent });
   257→   257→    }
   258→   258→  }
   259→   259→
   260→   260→  return sections;
   261→   261→}

   264→   264→  if (!Array.isArray(arr)) return [];
