    16→model NewsItem {
    17→  id               String     @id @default(cuid())
    18→  title            String
    19→  titleAr          String? // ترجمة عربية تلقائية
    20→  summary          String     @default("")
    21→  summaryAr        String? // ترجمة عربية تلقائية
    22→  content          String?    @default("") // المحتوى الكامل للخبر
    23→  contentAr        String? // المحتوى الكامل مترجم — null until filled by prefetchArticleContent
    24→  source           String     @default("")
    25→  sourceName       String?    @default("") // اسم المصدر الظاهر للمستخدم
    26→  isOfficialSource Boolean    @default(false) // V1050: true = مصدر رسمي (بنك مركزي، وزارة), false = مصدر صحفي
    27→  url              String     @default("") // originalUrl - رابط المصدر الخارجي
    28→  category         String     @default("اقتصاد كلي")
    29→  categoryId       String? // English category ID key (e.g. 'stocks', 'crypto', 'economy') — NEW for dual-column system
    30→  sentiment        String     @default("neutral")
    31→  sentimentScore   Int        @default(55)
    32→  impactLevel      String     @default("low")
    33→  impactScore      Int        @default(0) // V101: 0-100 importance score (economic + political + geopolitical impact)
    34→  originalLanguage String     @default("en")
    35→  locale           String     @default("ar") // 'ar' | 'en' — locale of the published content. Arabic pipeline → 'ar', English pipeline → 'en'
    36→  newsType         String     @default("live") // live | breaking | article
    37→  affectedAssets   String     @default("[]") // JSON array
    38→  aiAnalysis       String? // JSON from AI analysis
    39→  isPublished      Boolean    @default(false) // V38: FALSE until article is fully complete and published
    40→  isReady          Boolean    @default(false) // TRUE only when content is fully processed (titleAr + contentAr + aiAnalysis) — IRREVERSIBLE once true
    41→  processingStage  String     @default("fetched") // fetched → content_loaded → translated → analyzed → imaged
    42→  retryCount       Int        @default(0) // Number of failed processing attempts — used to re-process
    43→  rejectCount      Int        @default(0) // V118: Number of times AI filter rejected this article — after 3, no longer auto-reset
    44→  lastError        String? // Last error message from failed processing step
    45→  imageUrl         String?
    46→  generatedImage   String? // Base64 encoded AI-generated illustrative image (V42: always base64, never external URL)
    47→  slug             String? // URL-friendly slug for SEO (uniqueness enforced by @@unique([slug, locale]) below)
    48→  views            Int        @default(0) // Article view count
    49→  publishedAt      DateTime? // V42: When article was first published — set once, never changed
    50→  fetchedAt        DateTime   @default(now())
    51→  createdAt        DateTime   @default(now())
    52→  updatedAt        DateTime   @updatedAt
    53→  bookmarks        Bookmark[]
    54→  comments         Comment[]
    55→
    56→  @@unique([slug, locale]) // V256: Composite unique — same slug allowed in different locales
    57→  @@index([newsType, fetchedAt])
    58→  @@index([category])
    59→  @@index([categoryId])
    60→  @@index([locale])
    61→  @@index([sentiment])
    62→  @@index([slug])
    63→  @@index([url]) // For deduplication by originalUrl
    64→  @@index([url, locale]) // For locale-aware deduplication
    65→  @@index([isReady, fetchedAt]) // For homepage queries filtering ready articles
    66→  @@index([isReady, locale, fetchedAt]) // For locale-filtered homepage queries
    67→  @@index([processingStage])
    68→  @@index([impactScore]) // V101: For priority sorting in article picker
    69→  @@map("news_items")
    70→}

  1204→// AgencyEvent — News Agency Service (V1084)
  1205→// ═══════════════════════════════════════════════════════════════
  1206→// Independent news-writer agent. Fetches from official sources
  1207→// (SEC EDGAR, Fed RSS, World Bank), uses LLM to write original
  1208→// Arabic articles, publishes directly to NewsItem.
  1209→//
  1210→// This model is SEPARATE from the news pipeline. The agency service
  1211→// does NOT touch orchestrator/analyzer/imager/publisher. It reads
  1212→// from this table + writes directly to NewsItem.
  1213→// ═══════════════════════════════════════════════════════════════
  1214→
  1215→model AgencyEvent {
  1216→  id String @id @default(cuid())
  1217→
  1218→  // Source identification
  1219→  sourceId   String // 'SEC' | 'FRED' | 'FedRSS' | 'WorldBank'
  1220→  externalId String // ID from the source (e.g., SEC accession number, Fed press release ID)
  1221→  sourceName String? // Display name: "SEC EDGAR", "Federal Reserve"
  1222→  url        String  @default("") // Original URL of the event
  1223→
  1224→  // Event data (raw)
  1225→  eventType  String // 'filing' | 'press_release' | 'data_release' | 'speech'
  1226→  title      String // Original title from source
  1227→  rawContent String  @db.Text // Raw content extracted from source
  1228→  category   String? // For placeholder image: 'economy' | 'stocks' | 'crypto' | 'commodities' | 'forex' | 'central_banks'
  1229→
  1230→  // Draft (filled after LLM generation)
  1231→  draftTitle   String? @db.Text // Arabic title written by LLM
  1232→  draftBody    String? @db.Text // Arabic body written by LLM
  1233→  draftSummary String? // Arabic summary (2-3 sentences)
  1234→  llmProvider  String? // 'groq' | 'mistral' | 'bedrock' — which LLM wrote the draft
  1235→
  1236→  // Context from internal DB (filled at draft time)
  1237→  internalContext String? @db.Text // JSON: related economic_events, stock_analyses, geopolitical_risks
  1238→
  1239→  // Status tracking
  1240→  status     String  @default("fetched") // fetched → drafted → published | failed
  1241→  retryCount Int     @default(0)
  1242→  lastError  String? @db.Text
  1243→
  1244→  // Locale (for multi-locale support in V2)
  1245→  locale String @default("ar")
  1246→
  1247→  // Relations
  1248→  newsItemId  String? // FK to NewsItem after publish (manual, not Prisma relation to avoid coupling)
  1249→  publishedAt DateTime? // When actually published to NewsItem
  1250→
  1251→  // Timestamps
  1252→  fetchedAt DateTime @default(now())
  1253→  createdAt DateTime @default(now())
  1254→  updatedAt DateTime @updatedAt
  1255→
  1256→  @@unique([sourceId, externalId]) // Dedup: same event from same source = skip
  1257→  @@index([status, createdAt])

  1260→  @@index([publishedAt])
  1261→  @@index([locale])
